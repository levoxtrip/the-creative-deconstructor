```
#ifdef GL_ES
precision mediump float;
#endif

uniform vec2 u_resolution;
uniform vec2 u_mouse;
uniform float u_time;

float lineSDF(vec2 uv,vec2 a, vec2 b,float thickness){
    vec2 pa = uv -a;
    vec2 ba = b -a;
    
    float t = dot(ba,pa)/dot(ba,ba);
    
    t = clamp(t,0.0,1.0);
    
    vec2 closest = a + ba*t;
    
    return length(closest-uv)-thickness;
}

void main() {
    vec2 uv = gl_FragCoord.xy/u_resolution.xy;
    uv.x *= u_resolution.x/u_resolution.y;
	
    uv *= 2.0;
    float row = floor(uv.y);
    float isRowOdd = mod(row,2.0);
    uv = fract(uv);
	uv.x += isRowOdd*sin(u_time)*0.15;
    uv -= 0.5;
    float sdf = 1.0-smoothstep(0.0,0.02,lineSDF(uv,vec2(cos(u_time*uv.y)*0.5,-0.2),vec2(0.1,cos(u_time*(uv.y*sin(u_time)*0.04))*0.5),0.22));
    
    vec3 color = vec3(sdf);
    

    gl_FragColor = vec4(color,1.0);
}
```

```glsl
#ifdef GL_ES
precision mediump float;
#endif

uniform vec2 u_resolution;
uniform vec2 u_mouse;
uniform float u_time;

float lineSDF(vec2 uv,vec2 a, vec2 b,float thickness){
    vec2 pa = uv -a;
    vec2 ba = b -a;
    
    float t = dot(ba,pa)/dot(ba,ba);
    
    t = clamp(t,0.0,1.0);
    
    vec2 closest = a + ba*t;
    
    return length(closest-uv)-thickness;
}

void main() {
    vec2 uv = gl_FragCoord.xy/u_resolution.xy;
    uv.x *= u_resolution.x/u_resolution.y;
	float timeStepper = 2.+floor(mod(u_time,3.0));
    uv *= timeStepper;
    float row = floor(uv.y);
    float isRowOdd = mod(row,2.0);
    uv = fract(uv);
	uv.x += isRowOdd*sin(u_time)*0.15;
    uv -= 0.5;
    float sdf = 1.0-smoothstep(0.0,0.02,lineSDF(uv,vec2(cos(u_time*uv.y)*0.5,-0.2),vec2(0.1,cos(u_time*(uv.y*sin(u_time)*0.04))*0.5),0.22));
    
    vec3 color = vec3(sdf*tan(u_time-400.),sdf*cos(u_time+400.),sdf*sin(u_time));
    

    gl_FragColor = vec4(color,1.0);
}
```
