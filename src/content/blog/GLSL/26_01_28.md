Alrighty, this time I want to play with the length function.
```glsl
uniform vec2 u_resolution;
uniform vec2 u_mouse;
uniform float u_time;

float circleSDF(vec2 uv,vec2 center, float radius){
    float dist = length(uv-center);
    return length(dist) -radius;
    
    
}


void main() {
    vec2 uv = gl_FragCoord.xy/u_resolution.xy;
    uv.x *= u_resolution.x/u_resolution.y;
	vec2 iuv = floor(uv*5.0);
    uv = fract(uv*5.0);
    iuv = mod(iuv,2.);
    
    
    vec3 color = vec3(0.);
    float shape = 0.0;
    if(iuv.y != 0.0){
    	shape = 1.0-smoothstep(0.0,sin(u_time)*0.5,circleSDF(uv,vec2(0.5,sin(u_time)),0.3));    
    } else {
        shape = 1.0-step(0.0,circleSDF(uv-vec2(sin(u_time)*0.5,-0.270),vec2(0.5),0.3));    
    }
    color = vec3(shape);

    gl_FragColor = vec4(color,1.0);
}
