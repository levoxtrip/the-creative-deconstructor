```glsl
#ifdef GL_ES
precision mediump float;
#endif

uniform vec2 u_resolution;
uniform vec2 u_mouse;
uniform float u_time;

float lineSDF(vec2 uv, vec2 a, vec2 b, float thickness){
    vec2 pa = uv-a;
    vec2 ba = b-a;
    float t = dot(pa,ba)/dot(ba,ba);
    t = clamp(t,0.0,1.0);
    vec2 closest = a +t*ba;
    
    return length(closest-uv)-thickness;
    
}

void main() {
    vec2 uv = gl_FragCoord.xy/u_resolution.xy;
    uv.x *= u_resolution.x/u_resolution.y;
	uv -= 0.5;
    vec3 color = vec3(0.);
    float sin1 = sin(u_time)*0.25;;
    float sin2 = sin(u_time+300.)*0.25;
    float cos1 = cos(u_time-400.)*0.25;;
    float cos2 = cos(u_time+700.)*0.25;
    float shape = lineSDF(uv,vec2(sin1,cos1),vec2(sin2,cos2),0.05);
    float c = 1.0-shape*(tan(u_time)*5.);
    
    float sin21 = sin(u_time)*0.25;;
    float sin22 = sin(u_time+1300.)*0.5;
    float cos21 = cos(u_time-1400.)*0.5;;
    float cos22 = cos(u_time+1700.)*0.5;
    float shape2 = lineSDF(uv,vec2(sin21,cos22),vec2(sin22,cos21),0.1);
    float c2 = 1.0-shape2*(tan(u_time+400.)*5.);
    
    
    c = min(c,c2);
    vec3 col1 = mix(vec3(1.0,0.0,0.0),vec3(0.0,0.0,1.0),c);
    vec3 col2 = mix(vec3(.5,1.0,0.0),vec3(0.5,0.5,1.0),c2);
    
    color = mix(col1,col2,length(uv));;

    gl_FragColor = vec4(color,1.0);
}
```
