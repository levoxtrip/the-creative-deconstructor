```glsl
#ifdef GL_ES
precision mediump float;
#endif

uniform vec2 u_resolution;
uniform vec2 u_mouse;
uniform float u_time;

float rectSDF(vec2 uv, vec2 size){
    uv -= vec2(0.5);
    uv = abs(uv);
    float dy = uv.y-size.y;
    float dx = uv.x-size.x;
    
    return max(dx,dy);
    
    
    
}


void main() {
    vec2 st = gl_FragCoord.xy/u_resolution.xy;
    st.x *= u_resolution.x/u_resolution.y;
    st*= 5.0;
    float row = floor(st.y);
    float isOddRow = mod(row,2.0);
	
    
    st = fract(st);
    st.x += fract(0.2*u_time)*0.8;
    st+= isOddRow * sin(st+u_time)*0.5;
    
    float shape = rectSDF(st,vec2(0.250*(0.5+isOddRow*0.2),0.5+cos(-u_time)*0.230));
    
    float c = step(0.0,shape);
    vec3 color = vec3(c);
    color = vec3(color);

    gl_FragColor = vec4(color,1.0);
}
```
