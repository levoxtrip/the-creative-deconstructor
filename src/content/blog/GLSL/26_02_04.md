```glsl
uniform vec2 u_resolution;
uniform vec2 u_mouse;
uniform float u_time;

float rectSDF(vec2 uv, vec2 center, vec2 size){
    vec2 centered = uv -center;
    vec2 dist = abs(centered) -size *0.5;
    return max(dist.x,dist.y);
}

void main() {
    vec2 uv = gl_FragCoord.xy/u_resolution.xy;
    uv.x *= u_resolution.x/u_resolution.y;
	uv -= vec2(0.5);
    vec3 color = vec3(0.);
    
    float shape = rectSDF(uv,vec2(0.0),vec2(sin(u_time)*0.45+0.5,cos(u_time+400.)*0.45+0.5));
    float c = smoothstep(sin(u_time)*0.065,0.0,shape);
    
    vec3 col1 = vec3(tan(u_time),0.4,0.0);
    vec3 col2 = vec3(0.3,cos(100.0+u_time*0.8),0.8);
    vec3 col = mix(col1,col2,c);
	color = vec3(col);
    gl_FragColor = vec4(color,1.0);
}
```
