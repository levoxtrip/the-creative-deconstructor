Alrighty, this time I want to play with the length function.
```glsl
void main(){
    vec2 uv = gl_FragCoord.xy/u_resolution.xy;
    uv -= vec2(sin(u_time),tan(u_time));
	vec3 color = vec3(0.0);
    
    float dist = length(uv);
    float dist2 = length(4.*uv);
    float dist3 = length(sin(u_time)*uv);

    color = vec3(fract(u_time*0.2)*fract(dist2),1.0-fract(u_time*0.002)*dist*0.5,fract(u_time*0.2)*dist3*2.0);
	gl_FragColor = vec4(color,1.0);
}
```
