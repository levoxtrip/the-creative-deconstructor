
# 26-02-27
I experimented today with smooth boolean operations to create almost a metaball, lavalamp style.

```glsl
uniform vec2 u_resolution;
uniform vec2 u_mouse;
uniform float u_time;

float smoothMin(float a, float b, float k){
    float h = max(k-abs(a-b),0.0)/k;
    return min(a,b)-h*h*k*0.25;
    
    
}

float circleSDF(vec2 p, vec2 center,float radius){
    vec2 uv = p-center;
    return length(uv)-radius;
    
}

void main() {
    vec2 uv = gl_FragCoord.xy/u_resolution.xy;
    uv.x *= u_resolution.x/u_resolution.y;

    vec3 color = vec3(0.);
    
    float s1 = circleSDF(uv,vec2(0.5+0.3*sin(u_time+0.494)*cos(5000.+u_time*0.02),0.5+0.3*cos(-u_time)),0.3+sin(u_time)*0.1);
    float s2 = circleSDF(uv,vec2(0.5+0.3*cos(5000.+u_time+0.494)*sin(5.+u_time*0.02),0.5+0.3*-cos(-u_time)),0.3+sin(500.+u_time)*0.1);
    float sf = smoothstep(-0.232,0.1,smoothMin(s1,s2,0.3+cos(u_time)*.2));
    float sf2 = smoothstep(-0.3+sin(u_time)*0.15,1.340,smoothMin(s1,s2,0.300));
    vec3 c1 = vec3(0.5+sin(u_time*0.2),0.960,0.5+sin(500.+u_time*0.2));
    vec3 c2 = vec3(0.000,0.075,0.375);
    vec3 c3 = vec3(0.980,0.059,0.059);
    vec3 c4 = vec3(0.920,0.016,0.794);
    vec3 r = mix(c1,c2,sf);
    vec3 r2 = mix(c3,c4,sf2);
  	color = vec3(r.r,r2.g,mix(r.b,cos(u_time)*r2.b,sin(u_time)));

    gl_FragColor = vec4(color,1.0);
}
```
