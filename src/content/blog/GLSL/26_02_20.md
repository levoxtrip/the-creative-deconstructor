```glsl
uniform vec2 u_resolution;
uniform vec2 u_mouse;
uniform float u_time;

float lineSDF(vec2 p,vec2 center, vec2 a, vec2 b, float thickness){
    vec2 uv = p-center;
    vec2 pa = uv-a;
    vec2 ba = b-a;
    float t = dot(pa,ba)/dot(ba,ba);
    
    t = clamp(t,0.0,1.0);
    vec2 dist = a +t*ba;
    return length(uv-dist)-thickness;  
}

float rectSDF(vec2 p, vec2 center, vec2 size){
    vec2 uv = p-center;
    vec2 dist = abs(uv)-size;
    return length(max(dist,0.)) + min(max(dist.x,dist.y),0.);
    
}


void main() {
    vec2 uv = gl_FragCoord.xy/u_resolution.xy;
    uv.x *= u_resolution.x/u_resolution.y;
	
    float shape = lineSDF(uv,vec2(0.5),vec2(cos(u_time+sin(u_time))*0.3,-0.290),vec2(sin(u_time)*0.3,0.4),0.020);
    shape = 1.0-step(0.0,shape);
    float shape4 = lineSDF(uv,vec2(0.5),vec2(-0.4,sin(u_time)*0.3),vec2(0.3,cos(u_time)*0.4),0.02);
    shape4 = 1.0-step(0.0,shape4);
    float shape2 = rectSDF(uv,vec2(0.5),vec2(sin(u_time)*0.3+0.3));
    
    float shape3 = rectSDF(uv,vec2(0.5),vec2(cos(u_time)*0.3+0.3));
    shape2 = 1.0-step(0.0,shape2);
    shape3 = 1.0-step(0.0,shape3);
    vec3 color = vec3(shape+shape4);
    vec3 color2 = vec3(shape2,0.0,0.0);
    vec3 color3 = vec3(0.0,0.0,shape3);
    color2 = mix(color3,color2,0.5+0.5*sin(u_time*0.2));
	color += color2;
    gl_FragColor = vec4(color,1.0);
}
```
