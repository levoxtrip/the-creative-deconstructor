uniform vec2 u_resolution;
uniform vec2 u_mouse;
uniform float u_time;

float circleSDF(vec2 uv, vec2 center, float radius){
    vec2 centered = uv-center;
    return length(centered)-radius;
    
    
    
}

void main() {
    vec2 uv = gl_FragCoord.xy/u_resolution.xy;
    uv.x *= u_resolution.x/u_resolution.y;
	float s1 = 1.0-circleSDF(uv,vec2(0.5),0.3);
    
    float s2 = circleSDF(uv,vec2(cos(u_time)*sin(u_time*0.3)*0.5+0.6,sin(u_time)*(cos(u_time*0.2)*0.5+0.5)),0.1);
    
    float r = s1-s2;
    
    vec3 color = vec3(s1+r,r,s1-r);


    gl_FragColor = vec4(color,1.0);
}
